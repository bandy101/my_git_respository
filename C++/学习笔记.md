#十进制和二进制互相转换
void int_to_bin(int value,char *x,int index=0) {

	if (value / 2 <= 0)
		return;
	else
	{
		x[index] = value % 2;
		int_to_bin(value / 2, x, ++index);
	}
}
void bin_to_int(char *x, int *value,int e=0)
{
	if (e>=strlen(x)) return;
	*value = *value*2 + x[e] - '0';
	bin_to_int(x, value, ++e);
	/*
	char *k = "1000";
	bin_to_int(k, &value);##value =2*2*2 =8
	*/
}
